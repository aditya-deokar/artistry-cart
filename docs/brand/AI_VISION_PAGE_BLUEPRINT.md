POST /api/ai/generate-concept
  Body: { prompt, category?, style?, material? }

POST /api/ai/generate-from-product
  Body: { 
    productId: string,           # Base product to vary from
    modifications: string,        # What to change
    adjustments?: {               # Structured changes
      color?: string[],
      size?: { width, height, depth },
      material?: string[],
      style?: string
    }
  }
  Response: { 
    baseProduct: {...},           # Original product data
    variations: [...],            # AI-generated concepts
    estimatedPrice: number,       # Based on modification complexity
    suggestedArtisans: [...]      # Who made similar items
  }

# NEW: Visual search endpoint
POST /api/ai/visual-search
  Body: { 
    imageFile?: File,             # Uploaded image
    imageUrl?: string,            # Or URL
    filters?: {
      category?: string[],
      priceRange?: { min, max },
      inStock?: boolean,
      artisanId?: string
    },
    threshold?: number            # Min similarity (default 0.7)
  }
  Response: {
    matches: [                    # Database products
      {
        product: {...},
        similarityScore: number,  # 0-1 confidence
        visualFeatures: {         # What matched
          primaryColor: string,
          shape: string,
          style: string[]
        }
      }
    ],
    totalMatches: number,
    canGenerate: boolean          # If no good match, offer generation
  }

POST /api/ai/generate-from-image
  Body: { 
    imageFile?: File,
    imageUrl?: string,

  }

    analyzedFeatures: {           # AI's understanding
      category: string,
      style: string[],
      colors: string[],
      materials: string[],
      estimatedDimensions: {...}
    },
    concepts: [...],              # Generated variations

  }

POST /api/ai/refine-concept
  Body: { promptId, adjustments }
  Response: { concepts: [...] }

GET /api/ai/gallery
  Query: { category?, limit?, offset? }
  Response: { concepts: [...], hasMore }

POST /api/ai/save-concept
  Body: { conceptId, userId }
  Response: { saved: true }

GET /api/artisans/match
  Query: { conceptId }
  Response: { matches: [...] }

// useAIVision.ts
interface AIVisionState {

  // Text generation
  currentPrompt: string;
  generatedConcepts: Concept[];
  
  // Product variation (NEW!)
  selectedBaseProduct: Product | null;
  variationModifications: string;
  generatedVariations: Concept[];
  
  // Visual search (NEW!)
  uploadedImage: File | string | null;
  searchResults: ProductMatch[];
  visualSearchFilters: {
    category?: string[];
    priceRange?: { min: number; max: number };
    inStock?: boolean;
  };
  
  // Common state
  savedConcepts: Concept[];
  selectedConcept: Concept | null;
  matchedArtisans: Artisan[];
  isGenerating: boolean;
  isSearching: boolean;
  
  // Actions
  setMode: (mode) => void;
  searchByImage: (image: File | string, filters?) => Promise<void>;
  generateFromProduct: (productId: string, mods: string) => Promise<void>;
  sendToArtisans: (conceptId: string, artisanIds: string[]) => Promise<void>;
}
